from collections import defaultdict

import pandas as pd
import torch
from sentence_transformers import InputExample, SentenceTransformer, losses, util
from sklearn.preprocessing import LabelEncoder
from torch.utils.data import DataLoader

df = pd.read_csv("products.tab", sep="\t")

texts = df["text"].tolist()
cats = df["category"].tolist()

import random

# Get all texts and categories
texts = df["text"].tolist()
cats = df["category"].tolist()

# Sample similar pairs
num_similar = 20000

similar_pairs = []
texts_per_cat = defaultdict(list)
for t, c in zip(texts, cats):
    texts_per_cat[c].append(t)

for c, texts in texts_per_cat.items():
    for i in range(num_similar // len(texts_per_cat)):
        txt1, txt2 = random.sample(texts, 2)
        similar_pairs.append([txt1, txt2, 1])

# Sample dissimilar pairs
num_dissimilar = 20000
categories = list(texts_per_cat.keys())

dissimilar_pairs = []
for i in range(num_dissimilar):
    c1, c2 = random.sample(categories, 2)
    txt1 = random.choice(texts_per_cat[c1])
    txt2 = random.choice(texts_per_cat[c2])
    dissimilar_pairs.append([txt1, txt2, 0])

# Combine similar and dissimilar pairs
train_pairs = similar_pairs + dissimilar_pairs
random.shuffle(train_pairs)

# Convert to InputExamples
train_examples = [
    InputExample(texts=pair[:-1], label=float(pair[-1])) for pair in train_pairs
]

label_encoder = LabelEncoder()
num_cats = label_encoder.fit_transform(cats)
labels = list(set(num_cats.tolist()))
device = torch.device("cuda" if torch.cuda.is_available() else "cpu")

# Corrected the SentenceTransformer initialization, removing the incorrect arguments
model = SentenceTransformer("all-mpnet-base-v2")

model = model.to(device)

# DataLoader
train_dataloader = DataLoader(train_examples, shuffle=True, batch_size=16)

# Use triplet loss for similarity
train_loss = losses.CosineSimilarityLoss(model)

# Train the model
model.fit(
    train_objectives=[(train_dataloader, train_loss)], epochs=5, evaluation_steps=1000
)

# Save model
model.save("product_model.pt")

import numpy as np
from sklearn.preprocessing import MinMaxScaler
from tqdm import tqdm

def get_som_map(data, som_map, som_width=50, som_height=50, input_dim=768, num_iterations=5, learning_rate=0.1, influence_radius=5):
    # Initialize SOM weights

    # Normalize the data to the range [0, 1]
    scaler = MinMaxScaler()
    data = scaler.fit_transform(data)

    for epoch in tqdm(range(num_iterations)):
        for x in range(data.shape[0]):
            # Find best matching unit (BMU)
            diffs = som_map - data[x, :]
            dists = np.linalg.norm(diffs, axis=2)
            bmu_idx = np.unravel_index(np.argmin(dists), shape=(som_width, som_height))
            bmu_i, bmu_j = bmu_idx  # Unpack the tuple
            # Update SOM weights within the influence radius
            for i in range(max(0, bmu_i - influence_radius), min(som_width, bmu_i + influence_radius + 1)):
                for j in range(max(0, bmu_j - influence_radius), min(som_height, bmu_j + influence_radius + 1)):
                    # Calculate the Gaussian influence
                    distance_squared = (i - bmu_i)**2 + (j - bmu_j)**2
                    influence = 1 - distance_squared/influence_radius**2
                    som_map[i, j, :] += (learning_rate * influence * (data[x, :] - som_map[i, j, :]))

        # Decay learning rate
        learning_rate *= 0.9999

    return som_map, data, scaler

def compute_bmu_indices(som_map, data):
    bmu_indices = []

    for x in range(data.shape[0]):
        # Find best matching unit (BMU)
        diffs = data[x, :] - som_map
        dists = np.linalg.norm(diffs, axis=2)
        bmu_idx = np.unravel_index(np.argmin(dists), shape=(som_map.shape[0], som_map.shape[1]))
        bmu_i, bmu_j = bmu_idx  # Unpack the tuple
        bmu_indices.append((bmu_i, bmu_j))  # Store BMU indices

    return bmu_indices
